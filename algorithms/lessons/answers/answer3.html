var showStar = true; // 将 showStar 的值改为 true，使星星显示出来
var starCount = 3; // 将 starCount 的值改为 3，让页面上出现三颗星星
var starName = ["闪亮之星", "耀眼之星", "神秘之星"]; // 创建一个包含三颗星星名字的数组

这段代码通过修改 showStar 和 starCount 的值来控制星星的显示和数量。同时，starNames 数组被用来存储每颗星星的名称，这增加了代码的可读性和可管理性。
